24/// A set of emulated device types that can be accessed by a specific address range type.
25pub struct AxEmuDevices<R: DeviceAddrRange> {
26    emu_devices: Vec<Arc<dyn BaseDeviceOps<R>>>,
27}
28
29impl<R: DeviceAddrRange + 'static> AxEmuDevices<R> {
30    /// Creates a new [`AxEmuDevices`] instance.
31    pub fn new() -> Self {
32        Self {
33            emu_devices: Vec::new(),
34        }
35    }
36
37    /// Adds a device to the set.
38    pub fn add_dev(&mut self, dev: Arc<dyn BaseDeviceOps<R>>) {
39        self.emu_devices.push(dev);
40    }
41
42    // pub fn remove_dev(&mut self, ...)
43    //
44    // `remove_dev` seems to need something like `downcast-rs` to make sense. As it's not likely to
45    // be able to have a proper predicate to remove a device from the list without knowing the
46    // concrete type of the device.
47
48    /// Find a device by address.
49    pub fn find_dev(&self, addr: R::Addr) -> Option<Arc<dyn BaseDeviceOps<R>>> {
50        self.emu_devices
51            .iter()
52            .find(|&dev| dev.address_range().contains(addr))
53            .cloned()
54    }
55
56    /// Iterates over the devices in the set.
57    pub fn iter(&self) -> impl Iterator<Item = &Arc<dyn BaseDeviceOps<R>>> {
58        self.emu_devices.iter()
59    }
60
61    /// Iterates over the devices in the set mutably.
62    pub fn iter_mut(&mut self) -> impl Iterator<Item = &mut Arc<dyn BaseDeviceOps<R>>> {
63        self.emu_devices.iter_mut()
64    }
65}

81#[inline]
82fn log_device_io(
83    addr_type: &'static str,
84    addr: impl core::fmt::LowerHex,
85    addr_range: impl core::fmt::LowerHex,
86    read: bool,
87    width: AccessWidth,
88) {
89    let rw = if read { "read" } else { "write" };
90    trace!(
91        "emu_device {}: {} {:#x} in range {:#x} with width {:?}",
92        rw, addr_type, addr, addr_range, width
93    )
94}
95
96#[inline]
97fn panic_device_not_found(
98    addr_type: &'static str,
99    addr: impl core::fmt::LowerHex,
100    read: bool,
101    width: AccessWidth,
102) -> ! {
103    let rw = if read { "read" } else { "write" };
104    error!(
105        "emu_device {} failed: device not found for {} {:#x} with width {:?}",
106        rw, addr_type, addr, width
107    );
108    panic!("emu_device not found");
109}

126    /// According the emu_configs to init every  specific device
127    fn init(this: &mut Self, emu_configs: &Vec<EmulatedDeviceConfig>) {
128        for config in emu_configs {
129            match config.emu_type {
130                EmuDeviceType::InterruptController => {
131                    #[cfg(target_arch = "aarch64")]
132                    {
133                        this.add_mmio_dev(Arc::new(Vgic::new()));
134                    }
135                    #[cfg(not(target_arch = "aarch64"))]
136                    {
137                        warn!(
138                            "emu type: {} is not supported on this platform",
139                            config.emu_type
140                        );
141                    }
142                }
143                EmuDeviceType::GPPTRedistributor => {
144                    #[cfg(target_arch = "aarch64")]
145                    {
146                        const GPPT_GICR_ARG_ERR_MSG: &'static str =
147                            "expect 3 args for gppt redistributor (cpu_num, stride, pcpu_id)";
148
149                        let cpu_num = config
150                            .cfg_list
151                            .get(0)
152                            .copied()
153                            .expect(GPPT_GICR_ARG_ERR_MSG);
154                        let stride = config
155                            .cfg_list
156                            .get(1)
157                            .copied()
158                            .expect(GPPT_GICR_ARG_ERR_MSG);
159                        let pcpu_id = config
160                            .cfg_list
161                            .get(2)
162                            .copied()
163                            .expect(GPPT_GICR_ARG_ERR_MSG);
164
165                        for i in 0..cpu_num {
166                            let addr = config.base_gpa + i * stride;
167                            let size = config.length;
168                            this.add_mmio_dev(Arc::new(arm_vgic::v3::vgicr::VGicR::new(
169                                addr.into(),
170                                Some(size),
171                                pcpu_id + i,
172                            )));
173
174                            info!(
175                                "GPPT Redistributor initialized for vCPU {} with base GPA {:#x} and length {:#x}",
176                                i, addr, size
177                            );
178                        }
179                    }
180                    #[cfg(not(target_arch = "aarch64"))]
181                    {
182                        warn!(
183                            "emu type: {} is not supported on this platform",
184                            config.emu_type
185                        );
186                    }
187                }
188                EmuDeviceType::GPPTDistributor => {
189                    #[cfg(target_arch = "aarch64")]
190                    {
191                        this.add_mmio_dev(Arc::new(arm_vgic::v3::vgicd::VGicD::new(
192                            config.base_gpa.into(),
193                            Some(config.length),
194                        )));
195
196                        info!(
197                            "GPPT Distributor initialized with base GPA {:#x} and length {:#x}",
198                            config.base_gpa, config.length
199                        );
200                    }
201                    #[cfg(not(target_arch = "aarch64"))]
202                    {
203                        warn!(
204                            "emu type: {} is not supported on this platform",
205                            config.emu_type
206                        );
207                    }
208                }
209                EmuDeviceType::GPPTITS => {
210                    #[cfg(target_arch = "aarch64")]
211                    {
212                        let host_gits_base = config
213                            .cfg_list
214                            .get(0)
215                            .copied()
216                            .map(PhysAddr::from_usize)
217                            .expect("expect 1 arg for gppt its (host_gits_base)");
218
219                        this.add_mmio_dev(Arc::new(arm_vgic::v3::gits::Gits::new(
220                            config.base_gpa.into(),
221                            Some(config.length),
222                            host_gits_base,
223                            false,
224                        )));
225
226                        info!(
227                            "GPPT ITS initialized with base GPA {:#x} and length {:#x}, host GITS base {:#x}",
228                            config.base_gpa, config.length, host_gits_base
229                        );
230                    }
231                    #[cfg(not(target_arch = "aarch64"))]
232                    {
233                        warn!(
234                            "emu type: {} is not supported on this platform",
235                            config.emu_type
236                        );
237                    }
238                }
239                EmuDeviceType::IVCChannel => {
240                    if this.ivc_channel.is_none() {
241                        // Initialize the IVC channel range allocator
242                        this.ivc_channel = Some(Mutex::new(RangeAllocator::new(Range {
243                            start: config.base_gpa,
244                            end: config.base_gpa + config.length,
245                        })));
246                        info!(
247                            "IVCChannel initialized with base GPA {:#x} and length {:#x}",
248                            config.base_gpa, config.length
249                        );
250                    } else {
251                        warn!("IVCChannel already initialized, ignoring additional config");
252                    }
253                }
254                _ => {
255                    warn!(
256                        "Emulated device {}'s type {:?} is not supported yet",
257                        config.name, config.emu_type
258                    );
259                }
260            }
261        }
262    }

264    /// Allocates an IVC (Inter-VM Communication) channel of the specified size.
265    pub fn alloc_ivc_channel(&self, size: usize) -> AxResult<GuestPhysAddr> {
266        if size == 0 {
267            return ax_err!(InvalidInput, "Size must be greater than 0");
268        }
269        if !is_aligned_4k(size) {
270            return ax_err!(InvalidInput, "Size must be aligned to 4K");
271        }
272
273        if let Some(allocator) = &self.ivc_channel {
274            allocator
275                .lock()
276                .allocate_range(size)
277                .map_err(|e| {
278                    warn!("Failed to allocate IVC channel range: {:x?}", e);
279                    axerrno::ax_err_type!(NoMemory, "IVC channel allocation failed")
280                })
281                .map(|range| {
282                    debug!("Allocated IVC channel range: {:x?}", range);
283                    GuestPhysAddr::from_usize(range.start)
284                })
285        } else {
286            ax_err!(InvalidInput, "IVC channel not exists")
287        }
288    }
289
290    /// Releases an IVC channel at the specified address and size.
291    pub fn release_ivc_channel(&self, addr: GuestPhysAddr, size: usize) -> AxResult {
292        if size == 0 {
293            return ax_err!(InvalidInput, "Size must be greater than 0");
294        }
295        if !is_aligned_4k(size) {
296            return ax_err!(InvalidInput, "Size must be aligned to 4K");
297        }
298
299        if let Some(allocator) = &self.ivc_channel {
300            allocator
301                .lock()
302                .free_range(addr.as_usize()..addr.as_usize() + size);
303            Ok(())
304        } else {
305            ax_err!(InvalidInput, "IVC channel not exists")
306        }
307    }

374    /// Handle the MMIO read by GuestPhysAddr and data width, return the value of the guest want to read
375    pub fn handle_mmio_read(&self, addr: GuestPhysAddr, width: AccessWidth) -> AxResult<usize> {
376        if let Some(emu_dev) = self.find_mmio_dev(addr) {
377            log_device_io("mmio", addr, emu_dev.address_range(), true, width);
378
379            return emu_dev.handle_read(addr, width);
380        }
381        panic_device_not_found("mmio", addr, true, width);
382    }
